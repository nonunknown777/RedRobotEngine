29template <typename T>
30struct NonZeroChecker {
32 using exception_type = std::invalid_argument;

36 constexpr auto operator()(const T& v) -> decltype(!(v != static_cast<T>(0)), T{v})
37 {
38 if (!(v != static_cast<T>(0))) {
39 throw exception_type("value not non-zero");
40 }
41 return v;
42 }
