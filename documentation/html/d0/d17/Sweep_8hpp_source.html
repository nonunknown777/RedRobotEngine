114inline void Sweep::Advance0(const Real alpha) noexcept
115{
116 assert(IsValid(alpha));
117 assert(alpha >= 0);
118 assert(alpha < 1);
119 assert(alpha0 < 1);
120
121 const auto beta = (alpha - alpha0) / (1 - alpha0);
122 pos0 = GetPosition(pos0, pos1, beta);
123 alpha0 = alpha;
124}
125
126inline void Sweep::ResetAlpha0() noexcept
127{
128 alpha0 = 0;
