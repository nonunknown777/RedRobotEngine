9void MotionProperties::MoveKinematic(Vec3Arg inDeltaPosition, QuatArg inDeltaRotation, float inDeltaTime)
10{
11 JPH_ASSERT(BodyAccess::sCheckRights(BodyAccess::sVelocityAccess, BodyAccess::EAccess::ReadWrite));
12 JPH_ASSERT(BodyAccess::sCheckRights(BodyAccess::sPositionAccess, BodyAccess::EAccess::Read));
13 JPH_ASSERT(mCachedBodyType == EBodyType::RigidBody);
14 JPH_ASSERT(mCachedMotionType != EMotionType::Static);
15
16 // Calculate required linear velocity
17 mLinearVelocity = inDeltaPosition / inDeltaTime;
18
19 // Calculate required angular velocity
20 Vec3 axis;
21 float angle;
22 inDeltaRotation.GetAxisAngle(axis, angle);
23 mAngularVelocity = axis * (angle / inDeltaTime);
24}

76void MotionProperties::ApplyGyroscopicForceInternal(QuatArg inBodyRotation, float inDeltaTime)
77{
78 JPH_ASSERT(BodyAccess::sCheckRights(BodyAccess::sVelocityAccess, BodyAccess::EAccess::ReadWrite));
79 JPH_ASSERT(mCachedBodyType == EBodyType::RigidBody);
80 JPH_ASSERT(mCachedMotionType == EMotionType::Dynamic);
81
82 // Calculate local space inertia tensor (a diagonal in local space)
83 UVec4 is_zero = Vec3::sEquals(mInvInertiaDiagonal, Vec3::sZero());
84 Vec3 denominator = Vec3::sSelect(mInvInertiaDiagonal, Vec3::sReplicate(1.0f), is_zero);
85 Vec3 nominator = Vec3::sSelect(Vec3::sReplicate(1.0f), Vec3::sZero(), is_zero);
86 Vec3 local_inertia = nominator / denominator; // Avoid dividing by zero, inertia in this axis will be zero
87
88 // Calculate local space angular momentum
89 Quat inertia_space_to_world_space = inBodyRotation * mInertiaRotation;
90 Vec3 local_angular_velocity = inertia_space_to_world_space.Conjugated() * mAngularVelocity;
91 Vec3 local_momentum = local_inertia * local_angular_velocity;
92
93 // The gyroscopic force applies a torque: T = -w x I w where w is angular velocity and I the inertia tensor
94 // Calculate the new angular momentum by applying the gyroscopic force and make sure the new magnitude is the same as the old one
95 // to avoid introducing energy into the system due to the Euler step
96 Vec3 new_local_momentum = local_momentum - inDeltaTime * local_angular_velocity.Cross(local_momentum);
97 float new_local_momentum_len_sq = new_local_momentum.LengthSq();
98 new_local_momentum = new_local_momentum_len_sq > 0.0f? new_local_momentum * sqrt(local_momentum.LengthSq() / new_local_momentum_len_sq) : Vec3::sZero();
99
100 // Convert back to world space angular velocity
101 mAngularVelocity = inertia_space_to_world_space * (mInvInertiaDiagonal * new_local_momentum);
102}

104void MotionProperties::ApplyForceTorqueAndDragInternal(QuatArg inBodyRotation, Vec3Arg inGravity, float inDeltaTime)
105{
106 JPH_ASSERT(BodyAccess::sCheckRights(BodyAccess::sVelocityAccess, BodyAccess::EAccess::ReadWrite));
107 JPH_ASSERT(mCachedBodyType == EBodyType::RigidBody);
108 JPH_ASSERT(mCachedMotionType == EMotionType::Dynamic);
109
110 // Update linear velocity
111 mLinearVelocity = LockTranslation(mLinearVelocity + inDeltaTime * (mGravityFactor * inGravity + mInvMass * GetAccumulatedForce()));
112
113 // Update angular velocity
114 mAngularVelocity += inDeltaTime * MultiplyWorldSpaceInverseInertiaByVector(inBodyRotation, GetAccumulatedTorque());
115
116 // Linear damping: dv/dt = -c * v
117 // Solution: v(t) = v(0) * e^(-c * t) or v2 = v1 * e^(-c * dt)
118 // Taylor expansion of e^(-c * dt) = 1 - c * dt + ...
119 // Since dt is usually in the order of 1/60 and c is a low number too this approximation is good enough
120 mLinearVelocity *= max(0.0f, 1.0f - mLinearDamping * inDeltaTime);
121 mAngularVelocity *= max(0.0f, 1.0f - mAngularDamping * inDeltaTime);
122
123 // Clamp velocities
124 ClampLinearVelocity();
125 ClampAngularVelocity();
126}

128void MotionProperties::ResetSleepTestSpheres(const RVec3 *inPoints)
129{
130#ifdef JPH_DOUBLE_PRECISION
131 // Make spheres relative to the first point and initialize them to zero radius
132 DVec3 offset = inPoints[0];
133 offset.StoreDouble3(&mSleepTestOffset);
134 mSleepTestSpheres[0] = Sphere(Vec3::sZero(), 0.0f);
135 for (int i = 1; i < 3; ++i)
136 mSleepTestSpheres[i] = Sphere(Vec3(inPoints[i] - offset), 0.0f);
137#else
138 // Initialize the spheres to zero radius around the supplied points
139 for (int i = 0; i < 3; ++i)
140 mSleepTestSpheres[i] = Sphere(inPoints[i], 0.0f);
141#endif
142
143 mSleepTestTimer = 0.0f;
144}
