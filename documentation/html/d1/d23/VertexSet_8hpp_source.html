45 static Area GetDefaultMinSeparationSquared()
46 {
47 return sqrt(std::numeric_limits<Vec2::value_type>::min()) * SquareMeter;
48 }
49
51 explicit VertexSet(Area minSepSquared = GetDefaultMinSeparationSquared()):
52 m_minSepSquared{minSepSquared}
53 {
54 assert(minSepSquared >= 0_m2);
55 }
56
58 Area GetMinSeparationSquared() const noexcept { return m_minSepSquared; }
59
61 bool add(Length2 value)
62 {
63 if (find(value) != end())
64 {
65 return false;
66 }
67 m_elements.push_back(value);
68 return true;
69 }
70
72 void clear() noexcept

74 m_elements.clear();
75 }
76
78 std::size_t size() const noexcept
79 {
80 return ::playrho::detail::Size(m_elements);
81 }
82
84 const_pointer data() const { return ::playrho::detail::Data(m_elements); }
85
87 const_pointer begin() const { return data(); }
88
90 const_pointer end() const { return data() + size(); }
91
94 const_pointer find(Length2 value) const
95 {
96 // squaring anything smaller than the sqrt(std::numeric_limits<Vec2::data_type>::min())
97 // won't be reversible.
98 // i.e. won't obey the property that square(sqrt(a)) == a and sqrt(square(a)) == a.
99 return std::find_if(begin(), end(), [&](Length2 elem) {
100 // length squared must be large enough to have a reasonable enough unit vector.
101 return GetMagnitudeSquared(value - elem) <= m_minSepSquared;
102 });
103 }
104
106 Length2 operator[](std::size_t index) const noexcept
107 {
108 return m_elements[index];
109 }
110
111private:
112 std::vector<Length2> m_elements;
113 const Area m_minSepSquared;
114};
