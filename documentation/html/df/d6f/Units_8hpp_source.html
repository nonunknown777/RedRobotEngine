880template <class Unit>
881inline auto sqrt(const boost::units::quantity<Unit, Real>& q)
882{
883 using std::sqrt;
884 using quantity_type =
885 typename boost::units::root_typeof_helper<boost::units::quantity<Unit, Real>,
886 boost::units::static_rational<2>>::type;
887 using unit_type = typename quantity_type::unit_type;
888
889 return boost::units::quantity<unit_type, Real>::from_value(sqrt(q.value()));
890}

1013// Define division and multiplication templated operators in boost::units namespace since
1014// boost::units is the consistent namespace of operands for these and this aids with
1015// argument dependent lookup (ADL).
1016//
1017// Note that while boost::units already defines division and multiplication operator support,
1018// that's only for division or multiplication with the same type that the quantity is based
1019// on. For example when Real is float, Length{0.0f} * 2.0f is already supported but
1020// Length{0.0f} * 2 is not.
1021
1030template <class Dimension, typename X,
1031 typename = std::enable_if_t<
1032 playrho::IsArithmetic<X>::value && !std::is_same<X, playrho::Real>::value &&
1033 std::is_same<decltype(playrho::Real{} / X{}), playrho::Real>::value>>
1034constexpr auto operator/(quantity<Dimension, playrho::Real> lhs, X rhs)
1035{
1036 return lhs / playrho::Real(rhs);
1037}
1038
1039template <class Dimension, typename X,
1040 typename = std::enable_if_t<
1041 playrho::IsArithmetic<X>::value && !std::is_same<X, playrho::Real>::value &&
1042 std::is_same<decltype(X{} / playrho::Real{}), playrho::Real>::value>>
1043constexpr auto operator/(X lhs, quantity<Dimension, playrho::Real> rhs)
1044{

1046}
