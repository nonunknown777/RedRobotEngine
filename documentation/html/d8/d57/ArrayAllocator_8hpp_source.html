51 size_type GetIndex(value_type* ptr) const
52 {
53 const auto i = ptr - m_data.data();
54 return static_cast<size_type>(
55 ((i >= 0) && (static_cast<size_type>(i) < m_data.size())) ? i : -1);
56 }

62 template <class... Args>
63 size_type Allocate(Args&&... args)
64 {
65 if (!m_free.empty()) {
66 const auto index = m_free.back();
67 m_data[index] = value_type(std::forward<Args>(args)...);
68 m_free.pop_back();
69 return index;
70 }
71 const auto index = m_data.size();
72 m_data.emplace_back(std::forward<Args>(args)...);
73 return index;
74 }

80 size_type Allocate(const value_type& copy)
81 {
82 if (!m_free.empty()) {
83 const auto index = m_free.back();
84 m_data[index] = copy;
85 m_free.pop_back();
86 return index;
87 }
88 const auto index = m_data.size();
89 m_data.push_back(copy);
90 return index;
91 }

100 void Free(size_type index)
101 {
102 // only put index into free list if within size range
103 if (index >= m_data.size()) {
104 throw std::out_of_range("can't free given index");
105 }
106 m_data[index] = value_type{};
107 m_free.push_back(index);
108 }
109
114 bool FindFree(size_type index) const noexcept
115 {
116 for (const auto& element: m_free) {
117 if (element == index) {
118 return true;
119 }
120 }
121 return false;
122 }

174 void reserve(size_type value)

180 void clear() noexcept
181 {
182 m_data.clear();

184 }

186private:
