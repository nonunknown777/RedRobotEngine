173template <typename T, typename = std::enable_if_t<
174 IsIterable<T>::value && IsAddable<decltype(*begin(std::declval<T>()))>::value
175 > >
176inline auto Average(const T& span)
177{
178 using value_type = decltype(*begin(std::declval<T>()));
179
180 // Relies on C++11 zero initialization to zero initialize value_type.
181 // See: http://en.cppreference.com/w/cpp/language/zero_initialization
182 constexpr auto zero = value_type{};
183 assert(zero * Real{2} == zero);
184
185 // For C++17, switch from using std::accumulate to using std::reduce.
186 const auto sum = std::accumulate(begin(span), end(span), zero);
187 const auto count = std::max(size(span), std::size_t{1});
188 return sum / static_cast<Real>(count);
189}

236template <typename T>
237constexpr
238std::enable_if_t<std::is_floating_point<T>::value, bool> AlmostEqual(T x, T y, int ulp = 2)
239{
240 // From http://en.cppreference.com/w/cpp/types/numeric_limits/epsilon :
241 // "the machine epsilon has to be scaled to the magnitude of the values used
242 // and multiplied by the desired precision in ULPs (units in the last place)
243 // unless the result is subnormal".
244 // Where "subnormal" means almost zero.
245 //
246 return (abs(x - y) < (std::numeric_limits<T>::epsilon() * abs(x + y) * static_cast<T>(ulp)))
247 || AlmostZero(x - y);
248}

263template <typename T>
264inline auto ModuloViaTrunc(T dividend, T divisor) noexcept
265{
266 const auto quotient = dividend / divisor;
267 const auto integer = static_cast<T>(trunc(quotient));
268 const auto remainder = quotient - integer;
269 return remainder * divisor;
270}

275inline Angle GetNormalized(Angle value) noexcept
276{
277 constexpr auto oneRotationInRadians = Real{2 * Pi};
278 auto angleInRadians = Real{value / Radian};
279#if defined(NORMALIZE_ANGLE_VIA_FMOD)
280 // Note: std::fmod appears slower than std::trunc.
281 // See Benchmark ModuloViaFmod for data.
282 angleInRadians = ModuloViaFmod(angleInRadians, oneRotationInRadians);
283#else
284 // Note: std::trunc appears more than twice as fast as std::fmod.
285 // See Benchmark ModuloViaTrunc for data.
286 angleInRadians = ModuloViaTrunc(angleInRadians, oneRotationInRadians);
287#endif
288 if (angleInRadians > Pi)
289 {
290 // 190_deg becomes 190_deg - 360_deg = -170_deg
291 angleInRadians -= Pi * 2;
292 }
293 else if (angleInRadians < -Pi)
294 {
295 // -200_deg becomes -200_deg + 360_deg = 100_deg
296 angleInRadians += Pi * 2;
297 }
298 return angleInRadians * Radian;
299}

358template <typename T1, typename T2>
359constexpr auto Dot(const T1 a, const T2 b) noexcept
360{
361 static_assert(std::tuple_size<T1>::value == std::tuple_size<T2>::value,
362 "Dot only for same tuple-like sized types");
363 using VT1 = typename T1::value_type;
364 using VT2 = typename T2::value_type;
365 using OT = decltype(VT1{} * VT2{});
366 auto result = OT{};
367 const auto numElements = size(a);
368 for (auto i = decltype(numElements){0}; i < numElements; ++i)
369 {
370 result += a[i] * b[i];
371 }
372 return result;
373}

402template <class T1, class T2, std::enable_if_t<
403 std::tuple_size<T1>::value == 2 && std::tuple_size<T2>::value == 2, int> = 0>
404constexpr auto Cross(T1 a, T2 b) noexcept
405{
406 assert(isfinite(StripUnit(get<0>(a))));
407 assert(isfinite(StripUnit(get<1>(a))));
408 assert(isfinite(StripUnit(get<0>(b))));
409 assert(isfinite(StripUnit(get<1>(b))));
410
411 // Both vectors of same direction...
412 // If a = Vec2{1, 2} and b = Vec2{1, 2} then: a x b = 1 * 2 - 2 * 1 = 0.
413 // If a = Vec2{1, 2} and b = Vec2{2, 4} then: a x b = 1 * 4 - 2 * 2 = 0.
414 //
415 // Vectors at +/- 90 degrees of each other...
416 // If a = Vec2{1, 2} and b = Vec2{-2, 1} then: a x b = 1 * 1 - 2 * (-2) = 1 + 4 = 5.
417 // If a = Vec2{1, 2} and b = Vec2{2, -1} then: a x b = 1 * (-1) - 2 * 2 = -1 - 4 = -5.
418 //
419 // Vectors between 0 and 180 degrees of each other excluding 90 degrees...
420 // If a = Vec2{1, 2} and b = Vec2{-1, 2} then: a x b = 1 * 2 - 2 * (-1) = 2 + 2 = 4.
421 const auto minuend = get<0>(a) * get<1>(b);
422 const auto subtrahend = get<1>(a) * get<0>(b);
423 assert(isfinite(StripUnit(minuend)));
424 assert(isfinite(StripUnit(subtrahend)));
425 return minuend - subtrahend;
426}

434template <class T1, class T2, std::enable_if_t<
435 std::tuple_size<T1>::value == 3 && std::tuple_size<T2>::value == 3, int> = 0>
436constexpr auto Cross(T1 a, T2 b) noexcept
437{
438 assert(isfinite(get<0>(a)));
439 assert(isfinite(get<1>(a)));
440 assert(isfinite(get<2>(a)));
441 assert(isfinite(get<0>(b)));
442 assert(isfinite(get<1>(b)));
443 assert(isfinite(get<2>(b)));
444
445 using OT = decltype(get<0>(a) * get<0>(b));
446 return Vector<OT, 3>{
447 GetY(a) * GetZ(b) - GetZ(a) * GetY(b),
448 GetZ(a) * GetX(b) - GetX(a) * GetZ(b),
449 GetX(a) * GetY(b) - GetY(a) * GetX(b)
450 };
451}

455template <typename T, typename U>
456constexpr auto Solve(const Matrix22<U> mat, const Vector2<T> b) noexcept
457{
458 const auto cp = Cross(get<0>(mat), get<1>(mat));
459 const auto inverseCp = Real{1} / cp;
460 using OutType = decltype((U{} * T{}) / cp);
461 return (!AlmostZero(StripUnit(cp)))?
462 Vector2<OutType>{
463 (get<1>(mat)[1] * b[0] - get<1>(mat)[0] * b[1]) * inverseCp,
464 (get<0>(mat)[0] * b[1] - get<0>(mat)[1] * b[0]) * inverseCp
465 }: Vector2<OutType>{};
466}

469template <class IN_TYPE>
470constexpr auto Invert(const Matrix22<IN_TYPE> value) noexcept
471{
472 const auto cp = Cross(get<0>(value), get<1>(value));
473 using OutType = decltype(get<0>(value)[0] / cp);
474 const auto inverseCp = Real{1} / cp;
475 return (!AlmostZero(StripUnit(cp)))?
476 Matrix22<OutType>{
477 Vector2<OutType>{ get<1>(get<1>(value)) * inverseCp, -get<1>(get<0>(value)) * inverseCp},
478 Vector2<OutType>{-get<0>(get<1>(value)) * inverseCp, get<0>(get<0>(value)) * inverseCp}
479 }:
480 Matrix22<OutType>{};
481}

485constexpr Vec3 Solve33(const Mat33& mat, const Vec3 b) noexcept
486{
487 const auto dp = Dot(GetX(mat), Cross(GetY(mat), GetZ(mat)));
488 const auto det = (dp != 0)? Real{1} / dp: dp;
489 const auto x = det * Dot(b, Cross(GetY(mat), GetZ(mat)));
490 const auto y = det * Dot(GetX(mat), Cross(b, GetZ(mat)));
491 const auto z = det * Dot(GetX(mat), Cross(GetY(mat), b));
492 return Vec3{x, y, z};
493}

498template <typename T>
499constexpr T Solve22(const Mat33& mat, const T b) noexcept
500{
501 const auto cp = GetX(GetX(mat)) * GetY(GetY(mat)) - GetX(GetY(mat)) * GetY(GetX(mat));
502 const auto det = (cp != 0)? Real{1} / cp: cp;
503 const auto x = det * (GetY(GetY(mat)) * GetX(b) - GetX(GetY(mat)) * GetY(b));
504 const auto y = det * (GetX(GetX(mat)) * GetY(b) - GetY(GetX(mat)) * GetX(b));
505 return T{x, y};
506}

510constexpr Mat33 GetInverse22(const Mat33& value) noexcept
511{
512 const auto a = GetX(GetX(value)), b = GetX(GetY(value)), c = GetY(GetX(value)), d = GetY(GetY(value));
513 auto det = (a * d) - (b * c);
514 if (det != Real{0})
515 {
516 det = Real{1} / det;
517 }
518 return Mat33{Vec3{det * d, -det * c, Real{0}}, Vec3{-det * b, det * a, 0}, Vec3{0, 0, 0}};
519}

523constexpr Mat33 GetSymInverse33(const Mat33& value) noexcept
524{
525 auto det = Dot(GetX(value), Cross(GetY(value), GetZ(value)));
526 if (det != Real{0})
527 {
528 det = Real{1} / det;
529 }
530
531 const auto a11 = GetX(GetX(value)), a12 = GetX(GetY(value)), a13 = GetX(GetZ(value));
532 const auto a22 = GetY(GetY(value)), a23 = GetY(GetZ(value));
533 const auto a33 = GetZ(GetZ(value));
534
535 const auto ex_y = det * (a13 * a23 - a12 * a33);
536 const auto ey_z = det * (a13 * a12 - a11 * a23);
537 const auto ex_z = det * (a12 * a23 - a13 * a22);
538
539 return Mat33{
540 Vec3{det * (a22 * a33 - a23 * a23), ex_y, ex_z},
541 Vec3{ex_y, det * (a11 * a33 - a13 * a13), ey_z},
542 Vec3{ex_z, ey_z, det * (a11 * a22 - a12 * a12)}
543 };
544}

616template <typename T>
617constexpr T NextPowerOfTwo(T x)
618{
619 x |= (x >> 1u);
620 x |= (x >> 2u);
621 x |= (x >> 4u);
622 if constexpr (sizeof(T) >= 2u) {
623 x |= (x >> 8u);
624 }
625 if constexpr (sizeof(T) >= 4u) {
626 x |= (x >> 16u);
627 }
628 if constexpr (sizeof(T) >= 8u) {
629 x |= (x >> 32u);
630 }
631 if constexpr (sizeof(T) >= 16u) {
632 x |= (x >> 64u);
633 }
634 return x + 1;
635}

773template <class T>
774constexpr auto Rotate(const Vector2<T> vector, const UnitVec& angle) noexcept
775{
776 const auto newX = (GetX(angle) * GetX(vector)) - (GetY(angle) * GetY(vector));
777 const auto newY = (GetY(angle) * GetX(vector)) + (GetX(angle) * GetY(vector));
778 return Vector2<T>{newX, newY};
779}

788template <class T>
789constexpr auto InverseRotate(const Vector2<T> vector, const UnitVec& angle) noexcept
790{
791 const auto newX = (GetX(angle) * GetX(vector)) + (GetY(angle) * GetY(vector));
792 const auto newY = (GetX(angle) * GetY(vector)) - (GetY(angle) * GetX(vector));
793 return Vector2<T>{newX, newY};
794}

950constexpr auto GetReflectionMatrix(UnitVec axis)
951{
952 constexpr auto TupleSize = std::tuple_size<decltype(axis)>::value;
953 constexpr auto NumRows = TupleSize;
954 constexpr auto NumCols = TupleSize;
955 auto result = Matrix<Real, NumRows, NumCols>{};
956 for (auto row = decltype(NumRows){0}; row < NumRows; ++row)
957 {
958 for (auto col = decltype(NumCols){0}; col < NumCols; ++col)
959 {
960 result[row][col] = ((row == col)? Real{1}: Real{0}) - axis[row] * axis[col] * 2;
961 }
962 }
963 return result;
