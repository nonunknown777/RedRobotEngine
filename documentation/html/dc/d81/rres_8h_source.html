1/**********************************************************************************************
2*
3* rres v1.0 - A simple and easy-to-use file-format to package resources
4*
5* CONFIGURATION:
6*
7* #define RRES_IMPLEMENTATION
8* Generates the implementation of the library into the included file.
9* If not defined, the library is in header only mode and can be included in other headers
10* or source files without problems. But only ONE file should hold the implementation.
11*
12* FEATURES:
13*
14* - Multi-resource files: Some files could end-up generating multiple connected resources in
15* the rres output file (i.e TTF files could generate RRES_DATA_FONT_GLYPHS and RRES_DATA_IMAGE).
16* - File packaging as raw resource data: Avoid data processing and just package the file bytes.
17* - Per-file data compression/encryption: Configure compression/encription for every input file.
18* - Externally linked files: Package only the file path, to be loaded from external file when the
19* specific id is requested. WARNING: Be careful with path, it should be relative to application dir.
20* - Central Directory resource (optional): Create a central directory with the input filename relation
21* to the resource(s) id. This is the default option but it can be avoided; in that case, a header
22* file (.h) is generated with the file ids definitions.
23*
24* FILE STRUCTURE:
25*
26* rres files consist of a file header followed by a number of resource chunks.
27*
28* Optionally it can contain a Central Directory resource chunk (usually at the end) with the info
29* of all the files processed into the rres file.
30*
31* NOTE: Chunks count could not match files count, some processed files (i.e Font, Mesh)
32* could generate multiple chunks with the same id related by the rresResourceChunkInfo.nextOffset
33* Those chunks are loaded together when resource is loaded
34*
35* rresFileHeader (16 bytes)
36* Signature Id (4 bytes) // File signature id: 'rres'
37* Version (2 bytes) // Format version
38* Resource Count (2 bytes) // Number of resource chunks contained
39* CD Offset (4 bytes) // Central Directory offset (if available)
40* Reserved (4 bytes) // <reserved>
41*
42* rresResourceChunk[]
43* {
44* rresResourceChunkInfo (32 bytes)
45* Type (4 bytes) // Resource type (FourCC)
46* Id (4 bytes) // Resource identifier (CRC32 filename hash or custom)
47* Compressor (1 byte) // Data compression algorithm
48* Cipher (1 byte) // Data encryption algorithm
49* Flags (2 bytes) // Data flags (if required)
50* Data Packed Size (4 bytes) // Data packed size (compressed/encrypted + custom data appended)
51* Data Base Size (4 bytes) // Data base size (uncompressed/unencrypted)
52* Next Offset (4 bytes) // Next resource chunk offset (if required)
53* Reserved (4 bytes) // <reserved>
54* CRC32 (4 bytes) // Resource Data Chunk CRC32
55*
56* rresResourceChunkData (n bytes) // Packed data
57* Property Count (4 bytes) // Number of properties contained
58* Properties[] (4*i bytes) // Resource data required properties, depend on Type
59* Data (m bytes) // Resource data
60* }
61*
62* rresResourceChunk: RRES_DATA_DIRECTORY // Central directory (special resource chunk)
63* {
64* rresResourceChunkInfo (32 bytes)
65*
66* rresCentralDir (n bytes) // rresResourceChunkData
67* Entries Count (4 bytes) // Central directory entries count (files)
68* rresDirEntry[]
69* {
70* Id (4 bytes) // Resource id
71* Offset (4 bytes) // Resource global offset in file
72* reserved (4 bytes) // <reserved>
73* FileName Size (4 bytes) // Resource fileName size (NULL terminator and 4-bytes align padding considered)
74* FileName (m bytes) // Resource original fileName (NULL terminated and padded to 4-byte alignment)
75* }
76* }
77*
78* DESIGN DECISIONS / LIMITATIONS:
79*
80* - rres file maximum chunks: 65535 (16bit chunk count in rresFileHeader)
81* - rres file maximum size: 4GB (chunk offset and Central Directory Offset is 32bit, so it can not address more than 4GB
82* - Chunk search by ID is done one by one, starting at first chunk and accessed with fread() function
83* - Endianness: rres does not care about endianness, data is stored as desired by the host platform (most probably Little Endian)
84* Endianness won't affect chunk data but it will affect rresFileHeader and rresResourceChunkInfo
85* - CRC32 hash is used to to generate the rres file identifier from filename
86* There is a "small" probability of random collision (1 in 2^32 approx.) but considering
87* the chance of collision is related to the number of data inputs, not the size of the inputs, we assume that risk
88* Also note that CRC32 is not used as a security/cryptographic hash, just an identifier for the input file
89* - CRC32 hash is also used to detect chunk data corruption. CRC32 is smaller and computationally much less complex than MD5 or SHA1.
90* Using a hash function like MD5 is probably overkill for random error detection
91* - Central Directory rresDirEntry.fileName is NULL terminated and padded to 4-byte, rresDirEntry.fileNameSize considers the padding
92* - Compression and Encryption. rres supports chunks data compression and encryption, it provides two fields in the rresResourceChunkInfo to
93* note it, but in those cases is up to the user to implement the desired compressor/uncompressor and encryption/decryption mechanisms
94* In case of data encryption, it's recommended that any additional resource data (i.e. MAC) to be appended to data chunk and properly
95* noted in the packed data size field of rresResourceChunkInfo. Data compression should be applied before encryption.
96*
97* DEPENDENCIES:
98*
99* rres library dependencies has been keep to the minimum. It depends only some libc functionality:
100*
101* - stdlib.h: Required for memory allocation: malloc(), calloc(), free()
102* NOTE: Allocators can be redefined with macros RRES_MALLOC, RRES_CALLOC, RRES_FREE
103* - stdio.h: Required for file access functionality: FILE, fopen(), fseek(), fread(), fclose()
104* - string.h: Required for memory data management: memcpy(), memcmp()
105*
106* VERSION HISTORY:
107*
108* - 1.0 (12-May-2022): Implementation review for better alignment with rres specs
109* - 0.9 (28-Apr-2022): Initial implementation of rres specs

188// rres file header (16 bytes)
189typedef struct rresFileHeader {
190 unsigned char id[4]; // File identifier: rres
191 unsigned short version; // File version: 100 for version 1.0
192 unsigned short chunkCount; // Number of resource chunks in the file (MAX: 65535)
193 unsigned int cdOffset; // Central Directory offset in file (0 if not available)
194 unsigned int reserved; // <reserved>
195} rresFileHeader;
196
197// rres resource chunk info header (32 bytes)
198typedef struct rresResourceChunkInfo {
199 unsigned char type[4]; // Resource chunk type (FourCC)
200 unsigned int id; // Resource chunk identifier (generated from filename CRC32 hash)
201 unsigned char compType; // Data compression algorithm
202 unsigned char cipherType; // Data encription algorithm
203 unsigned short flags; // Data flags (if required)
204 unsigned int packedSize; // Data chunk size (compressed/encrypted + custom data appended)
205 unsigned int baseSize; // Data base size (uncompressed/unencrypted)
206 unsigned int nextOffset; // Next resource chunk global offset (if resource has multiple chunks)
207 unsigned int reserved; // <reserved>
208 unsigned int crc32; // Data chunk CRC32 (propCount + props[] + data)
209} rresResourceChunkInfo;
210
211// rres resource chunk data
212typedef struct rresResourceChunkData {
213 unsigned int propCount; // Resource chunk properties count
214 unsigned int *props; // Resource chunk properties
215 void *raw; // Resource chunk raw data
216} rresResourceChunkData;
217
218// rres resource chunk
219typedef struct rresResourceChunk {
220 rresResourceChunkInfo info; // Resource chunk info
221 rresResourceChunkData data; // Resource chunk packed data, contains propCount, props[] and raw data
222} rresResourceChunk;

224// rres resource multi
225// NOTE: It supports multiple resource chunks
226typedef struct rresResourceMulti {
227 unsigned int count; // Resource chunks count
228 rresResourceChunk *chunks; // Resource chunks
229} rresResourceMulti;

264// rres resource chunk data type
265// NOTE 1: Data type determines the properties and the data included in every chunk
266// NOTE 2: This enum defines the basic resource data types,
267// some input files could generate multiple resource chunks:
268// Fonts processed could generate (2) resource chunks:
269// - [FNTG] rres[0]: RRES_DATA_FONT_GLYPHS
270// - [IMGE] rres[1]: RRES_DATA_IMAGE
271//
272// Mesh processed could generate (n) resource chunks:
273// - [VRTX] rres[0]: RRES_DATA_VERTEX
274// ...
275// - [VRTX] rres[n]: RRES_DATA_VERTEX
276typedef enum rresResourceDataType {
277 RRES_DATA_NULL = 0, // FourCC: NULL - Reserved for empty chunks, no props/data
278 RRES_DATA_RAW = 1, // FourCC: RAWD - Raw file data, 4 properties
279 // props[0]:size (bytes)
280 // props[1]:extension01 (big-endian: ".png" = 0x2e706e67)
281 // props[2]:extension02 (additional part, extensions with +3 letters)
282 // props[3]:reserved
283 // data: raw bytes
284 RRES_DATA_TEXT = 2, // FourCC: TEXT - Text file data, 4 properties
285 // props[0]:size (bytes)
286 // props[1]:rresTextEncoding
287 // props[2]:rresCodeLang
288 // props[3]:cultureCode
289 // data: text
290 RRES_DATA_IMAGE = 3, // FourCC: IMGE - Image file data, 4 properties
291 // props[0]:width
292 // props[1]:height
293 // props[2]:rresPixelFormat
294 // props[3]:mipmaps
295 // data: pixels
296 RRES_DATA_WAVE = 4, // FourCC: WAVE - Audio file data, 4 properties
297 // props[0]:frameCount
298 // props[1]:sampleRate
299 // props[2]:sampleSize
300 // props[3]:channels
301 // data: samples
302 RRES_DATA_VERTEX = 5, // FourCC: VRTX - Vertex file data, 4 properties
303 // props[0]:vertexCount
304 // props[1]:rresVertexAttribute
305 // props[2]:componentCount
306 // props[3]:rresVertexFormat
307 // data: vertex
308 RRES_DATA_FONT_GLYPHS = 6, // FourCC: FNTG - Font glyphs info data, 4 properties
309 // props[0]:baseSize
310 // props[1]:glyphCount
311 // props[2]:glyphPadding
312 // props[3]:rresFontStyle
313 // data: rresFontGlyphInfo[0..glyphCount]
314 RRES_DATA_LINK = 99, // FourCC: LINK - External linked file, 1 property
315 // props[0]:size (bytes)
316 // data: filepath (as provided on input)
317 RRES_DATA_DIRECTORY = 100, // FourCC: CDIR - Central directory for input files
318 // props[0]:entryCount, 1 property
319 // data: rresDirEntry[0..entryCount]
320
321 // TODO: 2.0: Support resource package types (muti-resource)
322 // NOTE: They contains multiple rresResourceChunk in rresResourceData.raw
323 //RRES_DATA_PACK_FONT = 110, // FourCC: PFNT - Resources Pack: Font data, 1 property (2 resource chunks: RRES_DATA_GLYPHS, RRES_DATA_IMAGE)
324 // props[0]:chunkCount
325 //RRES_DATA_PACK_MESH = 120, // FourCC: PMSH - Resources Pack: Mesh data, 1 property (n resource chunks: RRES_DATA_VERTEX)
326 // props[0]:chunkCount
327
328 // TODO: Add additional resource data types if required (define props + data)
329
330} rresResourceDataType;

347// Encryption algoritms
348// Value required by rresResourceChunkInfo.cipherType
349// NOTE 1: This enum just lists some common data encryption algorithms for convenience,
350// The rres packer tool and the engine-specific library are responsible to implement the desired ones,
351// NOTE 2: Some encryption algorithm could require/generate additional data (seed, salt, nonce, MAC...)
352// in those cases, that extra data must be appended to the original encrypted message and added to the resource data chunk
353// NOTE 3: rresResourceChunkInfo.cipherType is a byte-size value, limited to [0..255]
354typedef enum rresEncryptionType {
355 RRES_CIPHER_NONE = 0, // No data encryption
356 RRES_CIPHER_XOR = 1, // XOR encryption, generic using 128bit key in blocks
357 RRES_CIPHER_DES = 10, // DES encryption
358 RRES_CIPHER_TDES = 11, // Triple DES encryption
359 RRES_CIPHER_IDEA = 20, // IDEA encryption
360 RRES_CIPHER_AES = 30, // AES (128bit or 256bit) encryption
361 RRES_CIPHER_AES_GCM = 31, // AES Galois/Counter Mode (Galois Message Authentification Code - GMAC)
362 RRES_CIPHER_XTEA = 40, // XTEA encryption
363 RRES_CIPHER_BLOWFISH = 50, // BLOWFISH encryption
364 RRES_CIPHER_RSA = 60, // RSA asymmetric encryption
365 RRES_CIPHER_SALSA20 = 70, // SALSA20 encryption
366 RRES_CIPHER_CHACHA20 = 71, // CHACHA20 encryption
367 RRES_CIPHER_XCHACHA20 = 72, // XCHACHA20 encryption
368 RRES_CIPHER_XCHACHA20_POLY1305 = 73, // XCHACHA20 with POLY1305 for message authentification (MAC)
369 // TODO: Add additional encryption algorithm if required
370} rresEncryptionType;

567// Load one resource chunk for provided id
568rresResourceChunk rresLoadResourceChunk(const char *fileName, int rresId)
569{
570 rresResourceChunk chunk = { 0 };
571
572 FILE *rresFile = fopen(fileName, "rb");
573
574 if (rresFile == NULL) RRES_LOG("RRES: WARNING: [%s] rres file could not be opened\n", fileName);
575 else
576 {
577 RRES_LOG("RRES: INFO: Loading resource from file: %s\n", fileName);
578
579 rresFileHeader header = { 0 };
580
581 // Read rres file header
582 fread(&header, sizeof(rresFileHeader), 1, rresFile);
583
584 // Verify file signature: "rres" and file version: 100
585 if (((header.id[0] == 'r') && (header.id[1] == 'r') && (header.id[2] == 'e') && (header.id[3] == 's')) && (header.version == 100))
586 {
587 bool found = false;
588
589 // Check all available chunks looking for the requested id
590 for (int i = 0; i < header.chunkCount; i++)
591 {
592 rresResourceChunkInfo info = { 0 };
593
594 // Read resource info header
595 fread(&info, sizeof(rresResourceChunkInfo), 1, rresFile);
596
597 // Check if resource id is the requested one
598 if (info.id == rresId)
599 {
600 found = true;
601
602 RRES_LOG("RRES: INFO: Found requested resource id: 0x%08x\n", info.id);
603 RRES_LOG("RRES: %c%c%c%c: Id: 0x%08x | Base size: %i | Packed size: %i\n", info.type[0], info.type[1], info.type[2], info.type[3], info.id, info.baseSize, info.packedSize);
604
605 // NOTE: We only load first matching id resource chunk found but
606 // we show a message if additional chunks are detected
607 if (info.nextOffset != 0) RRES_LOG("RRES: WARNING: Multiple linked resource chunks available for the provided id");
608
609 /*
610 // Variables required to check multiple chunks
611 int chunkCount = 0;
612 long currentFileOffset = ftell(rresFile); // Store current file position
613 rresResourceChunkInfo temp = info; // Temp info header to scan resource chunks
614
615 // Count all linked resource chunks checking temp.nextOffset
616 while (temp.nextOffset != 0)
617 {
618 fseek(rresFile, temp.nextOffset, SEEK_SET); // Jump to next linked resource
619 fread(&temp, sizeof(rresResourceChunkInfo), 1, rresFile); // Read next resource info header
620 chunkCount++;
621 }
622
623 fseek(rresFile, currentFileOffset, SEEK_SET); // Return to first resource chunk position
624 */
625
626 // Read and resource chunk from file data
627 // NOTE: Read data can be compressed/encrypted, it's up to the user library to manage decompression/decryption
628 void *data = RRES_MALLOC(info.packedSize); // Allocate enough memory to store resource data chunk
629 fread(data, info.packedSize, 1, rresFile); // Read data: propsCount + props[] + data (+additional_data)
630
631 // Get chunk.data properly organized (only if uncompressed/unencrypted)
632 chunk.data = rresLoadResourceChunkData(info, data);
633 chunk.info = info;
634
635 RRES_FREE(data);
636
637 break; // Resource id found and loaded, stop checking the file
638 }
639 else
640 {
641 // Skip required data size to read next resource info header
642 fseek(rresFile, info.packedSize, SEEK_CUR);
643 }
644 }
645
646 if (!found) RRES_LOG("RRES: WARNING: Requested resource not found: 0x%08x\n", rresId);
647 }
648 else RRES_LOG("RRES: WARNING: The provided file is not a valid rres file, file signature or version not valid\n");
649
650 fclose(rresFile);
651 }
652
653 return chunk;
654}

738 void *data = RRES_MALLOC(info.packedSize); // Allocate enough memory to store resource data chunk
739 fread(data, info.packedSize, 1, rresFile); // Read data: propsCount + props[] + data (+additional_data)
740
741 // Get chunk.data properly organized (only if uncompressed/unencrypted)
742 rres.chunks[i].data = rresLoadResourceChunkData(info, data);
743 rres.chunks[i].info = info;

745 RRES_FREE(data);

747 i++;
748 }
749
750 break; // Resource id found and loaded, stop checking the file

752 else
753 {
754 // Skip required data size to read next resource info header
755 fseek(rresFile, info.packedSize, SEEK_CUR);

757 }

760 }
761 else RRES_LOG("RRES: WARNING: The provided file is not a valid rres file, file signature or version not valid\n");
762
763 fclose(rresFile);

855// Load central directory data
856rresCentralDir rresLoadCentralDirectory(const char *fileName)
857{
858 rresCentralDir dir = { 0 };
859
860 FILE *rresFile = fopen(fileName, "rb");
861
862 if (rresFile != NULL)
863 {
864 rresFileHeader header = { 0 };
865
866 fread(&header, sizeof(rresFileHeader), 1, rresFile);
867
868 // Verify file signature: "rres", file version: 100
869 if (((header.id[0] == 'r') && (header.id[1] == 'r') && (header.id[2] == 'e') && (header.id[3] == 's')) && (header.version == 100))
870 {
871 // Check if there is a Central Directory available
872 if (header.cdOffset == 0) RRES_LOG("RRES: WARNING: CDIR: No central directory found\n");
873 else
874 {
875 rresResourceChunkInfo info = { 0 };
876
877 fseek(rresFile, header.cdOffset, SEEK_CUR); // Move to central directory position
878 fread(&info, sizeof(rresResourceChunkInfo), 1, rresFile); // Read resource info
879
880 // Verify resource type is CDIR
881 if ((info.type[0] == 'C') && (info.type[1] == 'D') && (info.type[2] == 'I') && (info.type[3] == 'R'))
882 {
883 RRES_LOG("RRES: CDIR: Central Directory found at offset: 0x%08x\n", header.cdOffset);
884
885 void *data = RRES_MALLOC(info.packedSize);
886 fread(data, info.packedSize, 1, rresFile);
887
888 // Load resource chunk data (central directory), data is uncompressed/unencrypted by default
889 rresResourceChunkData chunkData = rresLoadResourceChunkData(info, data);
890 RRES_FREE(data);
891
892 dir.count = chunkData.props[0]; // File entries count
893
894 RRES_LOG("RRES: CDIR: Central Directory file entries count: %i\n", dir.count);
895
896 unsigned char *ptr = (unsigned char *)chunkData.raw;
897 dir.entries = (rresDirEntry *)RRES_CALLOC(dir.count, sizeof(rresDirEntry));
898
899 for (unsigned int i = 0; i < dir.count; i++)
900 {
901 dir.entries[i].id = ((int *)ptr)[0]; // Resource id
902 dir.entries[i].offset = ((int *)ptr)[1]; // Resource offset in file
903 // NOTE: There is a reserved integer value before fileNameSize
904 dir.entries[i].fileNameSize = ((int *)ptr)[3]; // Resource fileName size
905
906 // Resource fileName, NULL terminated and 0-padded to 4-byte,
907 // fileNameSize considers NULL and padding
908 memcpy(dir.entries[i].fileName, ptr + 16, dir.entries[i].fileNameSize);
909
910 ptr += (16 + dir.entries[i].fileNameSize); // Move pointer for next entry
911 }
912
913 RRES_FREE(chunkData.props);
914 RRES_FREE(chunkData.raw);
915 }
916 }
917 }
918 else RRES_LOG("RRES: WARNING: The provided file is not a valid rres file, file signature or version not valid\n");
919
920 fclose(rresFile);
921 }
922
923 return dir;
924}

942 else if (memcmp(fourCC, "TEXT", 4) == 0) type = RRES_DATA_TEXT; // Text file data, byte data extracted from text file
943 else if (memcmp(fourCC, "IMGE", 4) == 0) type = RRES_DATA_IMAGE; // Image file data, pixel data extracted from image file
944 else if (memcmp(fourCC, "WAVE", 4) == 0) type = RRES_DATA_WAVE; // Audio file data, samples data extracted from audio file

946 else if (memcmp(fourCC, "FNTG", 4) == 0) type = RRES_DATA_FONT_GLYPHS; // Font glyphs info, generated from an input font file
947 else if (memcmp(fourCC, "LINK", 4) == 0) type = RRES_DATA_LINK; // External linked file, filepath as provided on file input
948 else if (memcmp(fourCC, "CDIR", 4) == 0) type = RRES_DATA_DIRECTORY; // Central directory for input files relation to resource chunks
949 }
950
951 /*
952 // Assign type (unsigned int) FourCC (char[4])
953 if ((fourCC[0] == 'N') && (fourCC[1] == 'U') && (fourCC[2] == 'L') && (fourCC[3] == 'L')) type = RRES_DATA_NULL; // NULL
954 if ((fourCC[0] == 'R') && (fourCC[1] == 'A') && (fourCC[2] == 'W') && (fourCC[3] == 'D')) type = RRES_DATA_RAW; // RAWD
955 else if ((fourCC[0] == 'T') && (fourCC[1] == 'E') && (fourCC[2] == 'X') && (fourCC[3] == 'T')) type = RRES_DATA_TEXT; // TEXT
956 else if ((fourCC[0] == 'I') && (fourCC[1] == 'M') && (fourCC[2] == 'G') && (fourCC[3] == 'E')) type = RRES_DATA_IMAGE; // IMGE
957 else if ((fourCC[0] == 'W') && (fourCC[1] == 'A') && (fourCC[2] == 'V') && (fourCC[3] == 'E')) type = RRES_DATA_WAVE; // WAVE
958 else if ((fourCC[0] == 'V') && (fourCC[1] == 'R') && (fourCC[2] == 'T') && (fourCC[3] == 'X')) type = RRES_DATA_VERTEX; // VRTX

1051// Load user resource chunk from resource packed data (as contained in .rres file)
1052// WARNING: Data can be compressed and/or encrypted, in those cases is up to the user to process it,
1053// and chunk.data.propCount = 0, chunk.data.props = NULL and chunk.data.raw contains all resource packed data
1054static rresResourceChunkData rresLoadResourceChunkData(rresResourceChunkInfo info, void *data)
1055{
1056 rresResourceChunkData chunkData = { 0 };
1057
1058 // CRC32 data validation, verify packed data is not corrupted
1059 unsigned int crc32 = rresComputeCRC32((unsigned char *)data, info.packedSize);
1060
1061 if ((rresGetDataType(info.type) != RRES_DATA_NULL) && (crc32 == info.crc32)) // Make sure chunk contains data and data is not corrupted
1062 {
1063 // Check if data chunk is compressed/encrypted to retrieve properties + data
1064 if ((info.compType == RRES_COMP_NONE) && (info.cipherType == RRES_CIPHER_NONE))
1065 {
1066 // Data is not compressed/encrypted (info.packedSize = info.baseSize)
1067 chunkData.propCount = ((unsigned int *)data)[0];
1068
1069 if (chunkData.propCount > 0)
1070 {
1071 chunkData.props = (unsigned int *)RRES_CALLOC(chunkData.propCount, sizeof(unsigned int));
1072 for (unsigned int i = 0; i < chunkData.propCount; i++) chunkData.props[i] = ((unsigned int *)data)[i + 1];
1073 }
1074
1075 int rawSize = info.baseSize - sizeof(int) - (chunkData.propCount*sizeof(int));
1076 chunkData.raw = RRES_MALLOC(rawSize);
1077 memcpy(chunkData.raw, ((unsigned char *)data) + sizeof(int) + (chunkData.propCount*sizeof(int)), rawSize);
1078 }
1079 else
1080 {
1081 // Data is compressed/encrypted
1082 // We just return the loaded resource packed data from .rres file,
1083 // it's up to the user to manage decompression/decryption on user library
1084 chunkData.raw = RRES_MALLOC(info.packedSize);
1085 memcpy(chunkData.raw, (unsigned char *)data, info.packedSize);
1086 }
1087 }
1088
1089 if (crc32 != info.crc32) RRES_LOG("RRES: WARNING: [ID %i] CRC32 does not match, data can be corrupted\n", info.id);
1090
1091 return chunkData;
1092}
