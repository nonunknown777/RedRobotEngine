112template <typename T>
113constexpr bool IsValid(const T& value) noexcept
114{
115 // Note: This is not necessarily a no-op!! But it is a "constexpr".
116 //
117 // From http://en.cppreference.com/w/cpp/numeric/math/isnan:
118 // "Another way to test if a floating-point value is NaN is
119 // to compare it with itself:
120 // bool is_nan(double x) { return x != x; }
121 //
122 // So for all T, for which isnan() is implemented, this should work
123 // correctly and quite usefully!
124 //
125 return value == value;
126}

276template <typename T>
277struct LexicographicalLess
278{
281 constexpr bool operator()(const T& lhs, const T& rhs) const
282 {
283 using std::less;
284 using ElementType = decltype(*begin(lhs));
285 return std::lexicographical_compare(begin(lhs), end(lhs), begin(rhs), end(rhs),
286 less<ElementType>{});

312template <typename T>
313struct LexicographicalLessEqual
314{
317 constexpr bool operator()(const T& lhs, const T& rhs) const
318 {
319 using std::mismatch;
320 using std::less;
321 using std::get;
322 using ElementType = decltype(*begin(lhs));
323 const auto lhsEnd = end(lhs);
324 const auto diff = mismatch(begin(lhs), lhsEnd, begin(rhs), end(rhs));
325 return (get<0>(diff) == lhsEnd) || less<ElementType>{}(*get<0>(diff), *get<1>(diff));
326 }
327};

353template <typename T, typename U>
354auto EraseFirst(T& container, const U& value) ->
355 decltype(container.erase(find(begin(container), end(container), value)) != end(container))
356{
357 const auto endIt = end(container);
358 const auto it = find(begin(container), endIt, value);
359 if (it != endIt) {
360 container.erase(it);
361 return true;
362 }
363 return false;
364}

370template <typename T, typename U>
371auto EraseAll(T& container, const U& value) ->
372 decltype(distance(container.erase(remove(begin(container), end(container), value), end(container)), end(container)))
373{
374 const auto itEnd = end(container);
375 const auto it = remove(begin(container), itEnd, value);
376 const auto count = distance(it, itEnd);
377 container.erase(it, itEnd);
378 return count;
379}
