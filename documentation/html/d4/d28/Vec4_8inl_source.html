17Vec4::Vec4(Vec3Arg inRHS, float inW)

30Vec4::Vec4(float inX, float inY, float inZ, float inW)
31{
32#if defined(JPH_USE_SSE)
33 mValue = _mm_set_ps(inW, inZ, inY, inX);
34#elif defined(JPH_USE_NEON)
35 uint32x2_t xy = vcreate_f32(static_cast<uint64>(*reinterpret_cast<uint32 *>(&inX)) | (static_cast<uint64>(*reinterpret_cast<uint32 *>(&inY)) << 32));
36 uint32x2_t zw = vcreate_f32(static_cast<uint64>(*reinterpret_cast<uint32* >(&inZ)) | (static_cast<uint64>(*reinterpret_cast<uint32 *>(&inW)) << 32));
37 mValue = vcombine_f32(xy, zw);
38#else
39 mF32[0] = inX;
40 mF32[1] = inY;
41 mF32[2] = inZ;
42 mF32[3] = inW;
43#endif
44}

46template<uint32 SwizzleX, uint32 SwizzleY, uint32 SwizzleZ, uint32 SwizzleW>
47Vec4 Vec4::Swizzle() const
48{
49 static_assert(SwizzleX <= 3, "SwizzleX template parameter out of range");
50 static_assert(SwizzleY <= 3, "SwizzleY template parameter out of range");
51 static_assert(SwizzleZ <= 3, "SwizzleZ template parameter out of range");
52 static_assert(SwizzleW <= 3, "SwizzleW template parameter out of range");
53
54#if defined(JPH_USE_SSE)
55 return _mm_shuffle_ps(mValue, mValue, _MM_SHUFFLE(SwizzleW, SwizzleZ, SwizzleY, SwizzleX));
56#elif defined(JPH_USE_NEON)
57 return JPH_NEON_SHUFFLE_F32x4(mValue, mValue, SwizzleX, SwizzleY, SwizzleZ, SwizzleW);
58#else
59 return Vec4(mF32[SwizzleX], mF32[SwizzleY], mF32[SwizzleZ], mF32[SwizzleW]);
60#endif
61}

112template <const int Scale>
113Vec4 Vec4::sGatherFloat4(const float *inBase, UVec4Arg inOffsets)
114{
115#if defined(JPH_USE_SSE)
116 #ifdef JPH_USE_AVX2
117 return _mm_i32gather_ps(inBase, inOffsets.mValue, Scale);
118 #else
119 const uint8 *base = reinterpret_cast<const uint8 *>(inBase);
120 Type x = _mm_load_ss(reinterpret_cast<const float *>(base + inOffsets.GetX() * Scale));
121 Type y = _mm_load_ss(reinterpret_cast<const float *>(base + inOffsets.GetY() * Scale));
122 Type xy = _mm_unpacklo_ps(x, y);
123 Type z = _mm_load_ss(reinterpret_cast<const float *>(base + inOffsets.GetZ() * Scale));
124 Type w = _mm_load_ss(reinterpret_cast<const float *>(base + inOffsets.GetW() * Scale));
125 Type zw = _mm_unpacklo_ps(z, w);
126 return _mm_movelh_ps(xy, zw);
127 #endif
128#else
129 const uint8 *base = reinterpret_cast<const uint8 *>(inBase);
130 float x = *reinterpret_cast<const float *>(base + inOffsets.GetX() * Scale);
131 float y = *reinterpret_cast<const float *>(base + inOffsets.GetY() * Scale);
132 float z = *reinterpret_cast<const float *>(base + inOffsets.GetZ() * Scale);
133 float w = *reinterpret_cast<const float *>(base + inOffsets.GetW() * Scale);
134 return Vec4(x, y, z, w);
135#endif
136}

301void Vec4::sSort4(Vec4 &ioValue, UVec4 &ioIndex)
302{
303 // Pass 1, test 1st vs 3rd, 2nd vs 4th
304 Vec4 v1 = ioValue.Swizzle<SWIZZLE_Z, SWIZZLE_W, SWIZZLE_X, SWIZZLE_Y>();
305 UVec4 i1 = ioIndex.Swizzle<SWIZZLE_Z, SWIZZLE_W, SWIZZLE_X, SWIZZLE_Y>();
306 UVec4 c1 = sLess(ioValue, v1).Swizzle<SWIZZLE_Z, SWIZZLE_W, SWIZZLE_Z, SWIZZLE_W>();
307 ioValue = sSelect(ioValue, v1, c1);
308 ioIndex = UVec4::sSelect(ioIndex, i1, c1);
309
310 // Pass 2, test 1st vs 2nd, 3rd vs 4th
311 Vec4 v2 = ioValue.Swizzle<SWIZZLE_Y, SWIZZLE_X, SWIZZLE_W, SWIZZLE_Z>();
312 UVec4 i2 = ioIndex.Swizzle<SWIZZLE_Y, SWIZZLE_X, SWIZZLE_W, SWIZZLE_Z>();
313 UVec4 c2 = sLess(ioValue, v2).Swizzle<SWIZZLE_Y, SWIZZLE_Y, SWIZZLE_W, SWIZZLE_W>();
314 ioValue = sSelect(ioValue, v2, c2);
315 ioIndex = UVec4::sSelect(ioIndex, i2, c2);
316
317 // Pass 3, test 2nd vs 3rd component
318 Vec4 v3 = ioValue.Swizzle<SWIZZLE_X, SWIZZLE_Z, SWIZZLE_Y, SWIZZLE_W>();
319 UVec4 i3 = ioIndex.Swizzle<SWIZZLE_X, SWIZZLE_Z, SWIZZLE_Y, SWIZZLE_W>();
320 UVec4 c3 = sLess(ioValue, v3).Swizzle<SWIZZLE_X, SWIZZLE_Z, SWIZZLE_Z, SWIZZLE_W>();
321 ioValue = sSelect(ioValue, v3, c3);
322 ioIndex = UVec4::sSelect(ioIndex, i3, c3);
323}

325void Vec4::sSort4Reverse(Vec4 &ioValue, UVec4 &ioIndex)
326{
327 // Pass 1, test 1st vs 3rd, 2nd vs 4th
328 Vec4 v1 = ioValue.Swizzle<SWIZZLE_Z, SWIZZLE_W, SWIZZLE_X, SWIZZLE_Y>();
329 UVec4 i1 = ioIndex.Swizzle<SWIZZLE_Z, SWIZZLE_W, SWIZZLE_X, SWIZZLE_Y>();

331 ioValue = sSelect(ioValue, v1, c1);

333
334 // Pass 2, test 1st vs 2nd, 3rd vs 4th
335 Vec4 v2 = ioValue.Swizzle<SWIZZLE_Y, SWIZZLE_X, SWIZZLE_W, SWIZZLE_Z>();
336 UVec4 i2 = ioIndex.Swizzle<SWIZZLE_Y, SWIZZLE_X, SWIZZLE_W, SWIZZLE_Z>();

364bool Vec4::IsNaN() const
365{
366#if defined(JPH_USE_AVX512)
367 return _mm_fpclass_ps_mask(mValue, 0b10000001) != 0;
368#elif defined(JPH_USE_SSE)
369 return _mm_movemask_ps(_mm_cmpunord_ps(mValue, mValue)) != 0;
370#elif defined(JPH_USE_NEON)
371 uint32x4_t is_equal = vceqq_f32(mValue, mValue); // If a number is not equal to itself it's a NaN
372 return vaddvq_u32(vshrq_n_u32(is_equal, 31)) != 4;
373#else
374 return isnan(mF32[0]) || isnan(mF32[1]) || isnan(mF32[2]) || isnan(mF32[3]);
375#endif
376}

766
767void Vec4::SinCos(Vec4 &outSin, Vec4 &outCos) const
768{
769 // Implementation based on sinf.c from the cephes library, combines sinf and cosf in a single function, changes octants to quadrants and vectorizes it
770 // Original implementation by Stephen L. Moshier (See: http://www.moshier.net/)
771
772 // Make argument positive and remember sign for sin only since cos is symmetric around x (highest bit of a float is the sign bit)
773 UVec4 sin_sign = UVec4::sAnd(ReinterpretAsInt(), UVec4::sReplicate(0x80000000U));
774 Vec4 x = Vec4::sXor(*this, sin_sign.ReinterpretAsFloat());
775
776 // x / (PI / 2) rounded to nearest int gives us the quadrant closest to x
777 UVec4 quadrant = (0.6366197723675814f * x + Vec4::sReplicate(0.5f)).ToInt();
778
779 // Make x relative to the closest quadrant.
780 // This does x = x - quadrant * PI / 2 using a two step Cody-Waite argument reduction.
781 // This improves the accuracy of the result by avoiding loss of significant bits in the subtraction.
782 // We start with x = x - quadrant * PI / 2, PI / 2 in hexadecimal notation is 0x3fc90fdb, we remove the lowest 16 bits to
783 // get 0x3fc90000 (= 1.5703125) this means we can now multiply with a number of up to 2^16 without losing any bits.
784 // This leaves us with: x = (x - quadrant * 1.5703125) - quadrant * (PI / 2 - 1.5703125).
785 // PI / 2 - 1.5703125 in hexadecimal is 0x39fdaa22, stripping the lowest 12 bits we get 0x39fda000 (= 0.0004837512969970703125)
786 // This leaves uw with: x = ((x - quadrant * 1.5703125) - quadrant * 0.0004837512969970703125) - quadrant * (PI / 2 - 1.5703125 - 0.0004837512969970703125)
787 // See: https://stackoverflow.com/questions/42455143/sine-cosine-modular-extended-precision-arithmetic
788 // After this we have x in the range [-PI / 4, PI / 4].
789 Vec4 float_quadrant = quadrant.ToFloat();
790 x = ((x - float_quadrant * 1.5703125f) - float_quadrant * 0.0004837512969970703125f) - float_quadrant * 7.549789948768648e-8f;
791
792 // Calculate x2 = x^2
793 Vec4 x2 = x * x;
794
795 // Taylor expansion:
796 // Cos(x) = 1 - x^2/2! + x^4/4! - x^6/6! + x^8/8! + ... = (((x2/8!- 1/6!) * x2 + 1/4!) * x2 - 1/2!) * x2 + 1
797 Vec4 taylor_cos = ((2.443315711809948e-5f * x2 - Vec4::sReplicate(1.388731625493765e-3f)) * x2 + Vec4::sReplicate(4.166664568298827e-2f)) * x2 * x2 - 0.5f * x2 + Vec4::sReplicate(1.0f);
798 // Sin(x) = x - x^3/3! + x^5/5! - x^7/7! + ... = ((-x2/7! + 1/5!) * x2 - 1/3!) * x2 * x + x
799 Vec4 taylor_sin = ((-1.9515295891e-4f * x2 + Vec4::sReplicate(8.3321608736e-3f)) * x2 - Vec4::sReplicate(1.6666654611e-1f)) * x2 * x + x;
800
801 // The lowest 2 bits of quadrant indicate the quadrant that we are in.
802 // Let x be the original input value and x' our value that has been mapped to the range [-PI / 4, PI / 4].
803 // since cos(x) = sin(x - PI / 2) and since we want to use the Taylor expansion as close as possible to 0,
804 // we can alternate between using the Taylor expansion for sin and cos according to the following table:
805 //
806 // quadrant sin(x) cos(x)
807 // XXX00b sin(x') cos(x')
808 // XXX01b cos(x') -sin(x')
809 // XXX10b -sin(x') -cos(x')
810 // XXX11b -cos(x') sin(x')
811 //
812 // So: sin_sign = bit2, cos_sign = bit1 ^ bit2, bit1 determines if we use sin or cos Taylor expansion
813 UVec4 bit1 = quadrant.LogicalShiftLeft<31>();
814 UVec4 bit2 = UVec4::sAnd(quadrant.LogicalShiftLeft<30>(), UVec4::sReplicate(0x80000000U));
815
816 // Select which one of the results is sin and which one is cos
817 Vec4 s = Vec4::sSelect(taylor_sin, taylor_cos, bit1);
818 Vec4 c = Vec4::sSelect(taylor_cos, taylor_sin, bit1);
819
820 // Update the signs
821 sin_sign = UVec4::sXor(sin_sign, bit2);
822 UVec4 cos_sign = UVec4::sXor(bit1, bit2);
823
824 // Correct the signs
825 outSin = Vec4::sXor(s, sin_sign.ReinterpretAsFloat());
826 outCos = Vec4::sXor(c, cos_sign.ReinterpretAsFloat());
827}

829Vec4 Vec4::Tan() const
830{
831 // Implementation based on tanf.c from the cephes library, see Vec4::SinCos for further details
832 // Original implementation by Stephen L. Moshier (See: http://www.moshier.net/)
833
834 // Make argument positive
835 UVec4 tan_sign = UVec4::sAnd(ReinterpretAsInt(), UVec4::sReplicate(0x80000000U));
836 Vec4 x = Vec4::sXor(*this, tan_sign.ReinterpretAsFloat());
837
838 // x / (PI / 2) rounded to nearest int gives us the quadrant closest to x
839 UVec4 quadrant = (0.6366197723675814f * x + Vec4::sReplicate(0.5f)).ToInt();
840
841 // Remap x to range [-PI / 4, PI / 4], see Vec4::SinCos
842 Vec4 float_quadrant = quadrant.ToFloat();
843 x = ((x - float_quadrant * 1.5703125f) - float_quadrant * 0.0004837512969970703125f) - float_quadrant * 7.549789948768648e-8f;
844
845 // Calculate x2 = x^2
846 Vec4 x2 = x * x;
847
848 // Roughly equivalent to the Taylor expansion:
849 // Tan(x) = x + x^3/3 + 2*x^5/15 + 17*x^7/315 + 62*x^9/2835 + ...
850 Vec4 tan =
851 (((((9.38540185543e-3f * x2 + Vec4::sReplicate(3.11992232697e-3f)) * x2 + Vec4::sReplicate(2.44301354525e-2f)) * x2
852 + Vec4::sReplicate(5.34112807005e-2f)) * x2 + Vec4::sReplicate(1.33387994085e-1f)) * x2 + Vec4::sReplicate(3.33331568548e-1f)) * x2 * x + x;
853
854 // For the 2nd and 4th quadrant we need to invert the value
855 UVec4 bit1 = quadrant.LogicalShiftLeft<31>();
856 tan = Vec4::sSelect(tan, Vec4::sReplicate(-1.0f) / (tan JPH_IF_FLOATING_POINT_EXCEPTIONS_ENABLED(+ Vec4::sReplicate(FLT_MIN))), bit1); // Add small epsilon to prevent div by zero, works because tan is always positive
857
858 // Put the sign back
859 return Vec4::sXor(tan, tan_sign.ReinterpretAsFloat());
860}
861
862Vec4 Vec4::ASin() const
863{
864 // Implementation based on asinf.c from the cephes library
865 // Original implementation by Stephen L. Moshier (See: http://www.moshier.net/)
866
867 // Make argument positive
868 UVec4 asin_sign = UVec4::sAnd(ReinterpretAsInt(), UVec4::sReplicate(0x80000000U));
869 Vec4 a = Vec4::sXor(*this, asin_sign.ReinterpretAsFloat());
870
871 // ASin is not defined outside the range [-1, 1] but it often happens that a value is slightly above 1 so we just clamp here
872 a = Vec4::sMin(a, Vec4::sReplicate(1.0f));
873
874 // When |x| <= 0.5 we use the asin approximation as is
875 Vec4 z1 = a * a;
876 Vec4 x1 = a;
877
878 // When |x| > 0.5 we use the identity asin(x) = PI / 2 - 2 * asin(sqrt((1 - x) / 2))
879 Vec4 z2 = 0.5f * (Vec4::sReplicate(1.0f) - a);
880 Vec4 x2 = z2.Sqrt();
881
882 // Select which of the two situations we have
883 UVec4 greater = Vec4::sGreater(a, Vec4::sReplicate(0.5f));
884 Vec4 z = Vec4::sSelect(z1, z2, greater);
885 Vec4 x = Vec4::sSelect(x1, x2, greater);
886
887 // Polynomial approximation of asin
888 z = ((((4.2163199048e-2f * z + Vec4::sReplicate(2.4181311049e-2f)) * z + Vec4::sReplicate(4.5470025998e-2f)) * z + Vec4::sReplicate(7.4953002686e-2f)) * z + Vec4::sReplicate(1.6666752422e-1f)) * z * x + x;
889
890 // If |x| > 0.5 we need to apply the remainder of the identity above
891 z = Vec4::sSelect(z, Vec4::sReplicate(0.5f * JPH_PI) - (z + z), greater);
892
893 // Put the sign back
894 return Vec4::sXor(z, asin_sign.ReinterpretAsFloat());
895}

897Vec4 Vec4::ACos() const

899 // Not the most accurate, but simple
900 return Vec4::sReplicate(0.5f * JPH_PI) - ASin();
901}
902
903Vec4 Vec4::ATan() const
904{
905 // Implementation based on atanf.c from the cephes library
906 // Original implementation by Stephen L. Moshier (See: http://www.moshier.net/)
907
908 // Make argument positive
909 UVec4 atan_sign = UVec4::sAnd(ReinterpretAsInt(), UVec4::sReplicate(0x80000000U));
910 Vec4 x = Vec4::sXor(*this, atan_sign.ReinterpretAsFloat());
911 Vec4 y = Vec4::sZero();
912
913 // If x > Tan(PI / 8)
914 UVec4 greater1 = Vec4::sGreater(x, Vec4::sReplicate(0.4142135623730950f));
915 Vec4 x1 = (x - Vec4::sReplicate(1.0f)) / (x + Vec4::sReplicate(1.0f));
916
917 // If x > Tan(3 * PI / 8)
918 UVec4 greater2 = Vec4::sGreater(x, Vec4::sReplicate(2.414213562373095f));
919 Vec4 x2 = Vec4::sReplicate(-1.0f) / (x JPH_IF_FLOATING_POINT_EXCEPTIONS_ENABLED(+ Vec4::sReplicate(FLT_MIN))); // Add small epsilon to prevent div by zero, works because x is always positive
920
921 // Apply first if
922 x = Vec4::sSelect(x, x1, greater1);
923 y = Vec4::sSelect(y, Vec4::sReplicate(0.25f * JPH_PI), greater1);
924
925 // Apply second if
926 x = Vec4::sSelect(x, x2, greater2);
927 y = Vec4::sSelect(y, Vec4::sReplicate(0.5f * JPH_PI), greater2);
928
929 // Polynomial approximation
930 Vec4 z = x * x;
931 y += (((8.05374449538e-2f * z - Vec4::sReplicate(1.38776856032e-1f)) * z + Vec4::sReplicate(1.99777106478e-1f)) * z - Vec4::sReplicate(3.33329491539e-1f)) * z * x + x;
932
933 // Put the sign back
934 return Vec4::sXor(y, atan_sign.ReinterpretAsFloat());
935}
936
937Vec4 Vec4::sATan2(Vec4Arg inY, Vec4Arg inX)
938{
939 UVec4 sign_mask = UVec4::sReplicate(0x80000000U);
940
941 // Determine absolute value and sign of y
942 UVec4 y_sign = UVec4::sAnd(inY.ReinterpretAsInt(), sign_mask);
943 Vec4 y_abs = Vec4::sXor(inY, y_sign.ReinterpretAsFloat());
944
945 // Determine absolute value and sign of x
946 UVec4 x_sign = UVec4::sAnd(inX.ReinterpretAsInt(), sign_mask);
947 Vec4 x_abs = Vec4::sXor(inX, x_sign.ReinterpretAsFloat());
948
949 // Always divide smallest / largest to avoid dividing by zero
950 UVec4 x_is_numerator = Vec4::sLess(x_abs, y_abs);
951 Vec4 numerator = Vec4::sSelect(y_abs, x_abs, x_is_numerator);
952 Vec4 denominator = Vec4::sSelect(x_abs, y_abs, x_is_numerator);
953 Vec4 atan = (numerator / denominator).ATan();
954
955 // If we calculated x / y instead of y / x the result is PI / 2 - result (note that this is true because we know the result is positive because the input was positive)
956 atan = Vec4::sSelect(atan, Vec4::sReplicate(0.5f * JPH_PI) - atan, x_is_numerator);
957
958 // Now we need to map to the correct quadrant
959 // x_sign y_sign result
960 // +1 +1 atan
961 // -1 +1 -atan + PI
962 // -1 -1 atan - PI
963 // +1 -1 -atan
964 // This can be written as: x_sign * y_sign * (atan - (x_sign < 0? PI : 0))
965 atan -= Vec4::sAnd(x_sign.ArithmeticShiftRight<31>().ReinterpretAsFloat(), Vec4::sReplicate(JPH_PI));
