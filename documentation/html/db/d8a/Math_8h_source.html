25inline float CenterAngleAroundZero(float inV)
26{
27 if (inV < -JPH_PI)
28 {
29 do
30 inV += 2.0f * JPH_PI;
31 while (inV < -JPH_PI);
32 }
33 else if (inV > JPH_PI)
34 {
35 do
36 inV -= 2.0f * JPH_PI;
37 while (inV > JPH_PI);
38 }
39 JPH_ASSERT(inV >= -JPH_PI && inV <= JPH_PI);
40 return inV;
41}

95inline uint CountTrailingZeros(uint32 inValue)
96{
97#if defined(JPH_CPU_X86) || defined(JPH_CPU_WASM)
98 #if defined(JPH_USE_TZCNT)
99 return _tzcnt_u32(inValue);
100 #elif defined(JPH_COMPILER_MSVC)
101 if (inValue == 0)
102 return 32;
103 unsigned long result;
104 _BitScanForward(&result, inValue);
105 return result;
106 #else
107 if (inValue == 0)
108 return 32;
109 return __builtin_ctz(inValue);
110 #endif
111#elif defined(JPH_CPU_ARM)
112 #if defined(JPH_COMPILER_MSVC)
113 if (inValue == 0)
114 return 32;
115 unsigned long result;
116 _BitScanForward(&result, inValue);
117 return result;
118 #else
119 return __builtin_clz(__builtin_bitreverse32(inValue));
120 #endif
121#else
122 #error Undefined
123#endif
124}

127inline uint CountLeadingZeros(uint32 inValue)
128{

132 #elif defined(JPH_COMPILER_MSVC)
133 if (inValue == 0)

144 #if defined(JPH_COMPILER_MSVC)

155inline uint CountBits(uint32 inValue)
156{
157#if defined(JPH_COMPILER_CLANG) || defined(JPH_COMPILER_GCC)
158 return __builtin_popcount(inValue);
159#elif defined(JPH_COMPILER_MSVC)
160 #if defined(JPH_USE_SSE4_2)
161 return _mm_popcnt_u32(inValue);
162 #elif defined(JPH_USE_NEON) && (_MSC_VER >= 1930) // _CountOneBits not available on MSVC2019
163 return _CountOneBits(inValue);
164 #else
165 inValue = inValue - ((inValue >> 1) & 0x55555555);
166 inValue = (inValue & 0x33333333) + ((inValue >> 2) & 0x33333333);
167 inValue = (inValue + (inValue >> 4)) & 0x0F0F0F0F;
168 return (inValue * 0x01010101) >> 24;
169 #endif
170#else
171 #error Undefined
172#endif
173}

176inline uint32 GetNextPowerOf2(uint32 inValue)
177{
178 return inValue <= 1? uint32(1) : uint32(1) << (32 - CountLeadingZeros(inValue - 1));

181// Simple implementation of C++20 std::bit_cast (unfortunately not constexpr)
182template <class To, class From>
183JPH_INLINE To BitCast(const From &inValue)
184{
185 static_assert(std::is_trivially_constructible_v<To>);
186 static_assert(sizeof(From) == sizeof(To));
187
188 union FromTo
189 {
190 To mTo;
191 From mFrom;
192 };
193
194 FromTo convert;
195 convert.mFrom = inValue;
196 return convert.mTo;
197}
