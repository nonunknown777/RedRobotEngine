144template <typename T, std::size_t M, std::size_t N>
145constexpr
146auto operator+ (const Matrix<T, M, N>& lhs, const Matrix<T, M, N>& rhs) noexcept
147{
148 auto result = Matrix<T, M, N>{};
149 for (auto m = decltype(M){0}; m < M; ++m)
150 {
151 for (auto n = decltype(N){0}; n < N; ++n)
152 {
153 result[m][n] = lhs[m][n] + rhs[m][n];
154 }
155 }
156 return result;
157}

161template <typename T, std::size_t M, std::size_t N>
162constexpr
163auto operator- (const Matrix<T, M, N>& lhs, const Matrix<T, M, N>& rhs) noexcept
164{
165 auto result = Matrix<T, M, N>{};
166 for (auto m = decltype(M){0}; m < M; ++m)
167 {
168 for (auto n = decltype(N){0}; n < N; ++n)
169 {
170 result[m][n] = lhs[m][n] - rhs[m][n];
171 }
172 }
173 return result;
174}
