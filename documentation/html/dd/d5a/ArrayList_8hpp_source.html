38template <typename VALUE_TYPE, std::size_t MAXSIZE, typename SIZE_TYPE = std::size_t>
39class ArrayList
40{

46 using value_type = VALUE_TYPE;

60 constexpr ArrayList() noexcept
61 {
62 // Intentionally empty.
63 // Note that defaulting this method instead of writing it out here,

67 template <std::size_t COPY_MAXSIZE, typename COPY_SIZE_TYPE,
68 typename = std::enable_if_t<COPY_MAXSIZE <= MAXSIZE>>
69 constexpr explicit ArrayList(const ArrayList<VALUE_TYPE, COPY_MAXSIZE, SIZE_TYPE>& copy)
70 : m_size{size(copy)}, m_elements{data(copy)}
71 {
72 // Intentionally empty
73 }
74
76 template <std::size_t COPY_MAXSIZE, typename COPY_SIZE_TYPE,
77 typename = std::enable_if_t<COPY_MAXSIZE <= MAXSIZE>>
78 ArrayList& operator=(const ArrayList<VALUE_TYPE, COPY_MAXSIZE, COPY_SIZE_TYPE>& copy)
79 {
80 m_size = static_cast<SIZE_TYPE>(size(copy));
81 m_elements = data(copy);
82 return *this;
83 }

85 template <std::size_t SIZE, typename = std::enable_if_t<SIZE <= MAXSIZE>>
86 explicit ArrayList(value_type (&value)[SIZE]) noexcept
87 {
88 for (auto&& elem : value) {
89 push_back(elem);
90 }
91 }
92
93 ArrayList(std::initializer_list<value_type> list)
94 {
95 for (auto&& elem : list) {
96 push_back(elem);
97 }
98 }
99
100 constexpr ArrayList& Append(const value_type& value)
101 {
102 push_back(value);
103 return *this;
104 }
105
106 constexpr void push_back(const value_type& value) noexcept
107 {
108 assert(m_size < MAXSIZE);
109 m_elements[m_size] = value;
110 ++m_size;
111 }
112
113 void size(size_type value) noexcept
114 {
115 assert(value <= MAXSIZE);
116 m_size = value;
117 }
118
119 void clear() noexcept
120 {
121 m_size = 0;
122 }
123
124 bool empty() const noexcept
125 {
126 return m_size == 0;
127 }
128
129 bool add(value_type value) noexcept
130 {
131 if (m_size < MAXSIZE) {
132 m_elements[m_size] = value;
133 ++m_size;
134 return true;
135 }
136 return false;
137 }
138
139 reference operator[](size_type index) noexcept
140 {
141 assert(index < MAXSIZE);
142 return m_elements[index];
143 }
144
145 constexpr const_reference operator[](size_type index) const noexcept
146 {
147 assert(index < MAXSIZE);
148 return m_elements[index];
149 }
150
155 constexpr size_type size() const noexcept
156 {
157 return m_size;
158 }
159
162 constexpr size_type max_size() const noexcept
163 {
164 return MAXSIZE;
165 }
166
167 auto data() const noexcept
168 {
169 return m_elements.data();
170 }
171
172 pointer begin() noexcept
173 {
174 return m_elements.data();
175 }
176 pointer end() noexcept
177 {
178 return m_elements.data() + m_size;
179 }
180
181 const_pointer begin() const noexcept
182 {
183 return m_elements.data();
184 }
185 const_pointer end() const noexcept
186 {
187 return m_elements.data() + m_size;
188 }
189
190private:
191 size_type m_size = size_type{0};
192 std::array<value_type, MAXSIZE> m_elements;
193};
