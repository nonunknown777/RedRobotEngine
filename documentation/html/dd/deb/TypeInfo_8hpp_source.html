47template <typename T>
48std::string TypeNameAsString()
49{
50 // Ideally return string unique to the type T...
51#if defined(__clang__)
52 // Use __PRETTY_FUNCTION__. **Note that despite its appearance, this is an identifier; it's not a macro**!
53 // template <typename T> string Name() { return string{__PRETTY_FUNCTION__}; }
54 // enum class Fruit {APPLE, PEAR};
55 // std::cout << Name<Fruit>() << '\n';
56 // produces: std::string Name() [T = Fruit]
57 return std::regex_replace(__PRETTY_FUNCTION__, std::regex(".*T = (.*)\\].*"), "$1");
58#elif defined(__GNUC__)
59 // Use __PRETTY_FUNCTION__. **Note that despite its appearance, this is an identifier; it's not a macro**!
60 // template <typename T> string Name() { return string{__PRETTY_FUNCTION__}; }
61 // enum class Fruit {APPLE, PEAR};
62 // std::cout << Name<Fruit>() << '\n';
63 // produces: std::string Name() [with T = Fruit; std::string = std::__cxx11::basic_string<char>]
64 return std::regex_replace(__PRETTY_FUNCTION__, std::regex(".*T = (.*);.*"), "$1");
65#elif defined(__FUNCSIG__)
66 // Assume this is Microsoft Visual C++ or compatible compiler and format.
67 // enum class Fruit {APPLE, PEAR};
68 // std::cout << Name<Fruit>() << '\n';
69 // produces:
70 // class std::basic_string<char,struct std::char_traits<char>,class std::allocator<char> > __cdecl Name<enum Fruit>(void)
71 return std::regex_replace(__FUNCSIG__, std::regex(".* __cdecl [^<]+<(.*)>\\(void\\)"), "$1");
72#else
73 return {typeid(T).name()}; // not demangled but still should be unique and non-empty.
74#endif
75}

120class TypeID
121{
122public:
125 TypeID() noexcept = default;
126
128 constexpr const char* GetName() const noexcept
129 {
130 return *m_name;
131 }
132
134 inline friend bool operator==(const TypeID& lhs, const TypeID& rhs) noexcept
135 {
136 return (lhs.m_name == rhs.m_name) || (std::strcmp(*lhs.m_name, *rhs.m_name) == 0);
137 }
138
140 inline friend bool operator!=(const TypeID& lhs, const TypeID& rhs) noexcept
141 {
142 return !(lhs == rhs);
143 }
144
147 inline friend bool operator<(const TypeID& lhs, const TypeID& rhs) noexcept
148 {
149 return (lhs.m_name != rhs.m_name) && (std::strcmp(*lhs.m_name, *rhs.m_name) < 0);
150 }
151
154 inline friend bool operator<=(const TypeID& lhs, const TypeID& rhs) noexcept
155 {
156 return (lhs.m_name == rhs.m_name) || (std::strcmp(*lhs.m_name, *rhs.m_name) <= 0);
157 }
158
161 inline friend bool operator>(const TypeID& lhs, const TypeID& rhs) noexcept
162 {
163 return !(lhs <= rhs);
164 }
165
168 inline friend bool operator>=(const TypeID& lhs, const TypeID& rhs) noexcept
169 {
170 return !(lhs < rhs);
171 }

173 template <typename T>

175
176 template <typename T>
177 friend TypeID GetTypeID(const T&) noexcept;
178
179private:

181 {
182 // Intentionally empty.
183 }
