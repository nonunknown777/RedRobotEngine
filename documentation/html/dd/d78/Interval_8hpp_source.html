181 constexpr Interval& Expand(const value_type& v) noexcept
182 {

201 constexpr Interval& ExpandEqually(const NonNegative<value_type>& v) noexcept
202 {
203 const auto amount = value_type{v};
204 m_min -= amount;
205 m_max += amount;
206 return *this;
207 }

250template <typename T>
251constexpr bool IsIntersecting(const Interval<T>& a, const Interval<T>& b) noexcept
252{
253 const auto maxOfMins = std::max(a.GetMin(), b.GetMin());
254 const auto minOfMaxs = std::min(a.GetMax(), b.GetMax());
255 return minOfMaxs >= maxOfMins;
